                for (var emptyCell = 1; emptyCell <= nunberRow - i; emptyCell++)

                for (var j = 0; j <= i; j++)

                    Console.Write($"{c,8}");
                    //Console.Write(c >= 100 ? (" ") : c >= 10 ? ("  ") : ("   "));
                }
                Console.Write("\n");
            }
            Console.ReadKey();
